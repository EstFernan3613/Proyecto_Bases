7. Promedio de notas de los empleados de Colombia.

SELECT AVG(articulo.calificacion) AS promedio_notas
FROM empleado
JOIN articulo ON empleado.Id_empleado = articulo.Id_empleado
WHERE empleado.Id_pais = '3';
